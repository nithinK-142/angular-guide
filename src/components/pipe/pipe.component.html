{{ today | date : "fullDate" }}

{{ today | date : "dd-MM-yy" }}

{{ today | date : "dd-MM-yyyy" }}

{{ today | date : "dd-MMM-yy" }}

{{ today | date : "hh:mm:ss" }}

Currency Pipe

{{ 123.45 | currency : "INR" }}

Percent Pipe

Uppercase, Lowercase, and Titlecase Pipes

{{ "hello world" | titlecase }}

{{ "Hello World" | slice : 6 : 11 }}
